<script>
(function() {
  // 快速导航切换
  const navItems = document.querySelectorAll('.ai-nav-item');
  const sections = document.querySelectorAll('.ai-section');

  navItems.forEach(item => {
    item.addEventListener('click', function() {
      const target = this.dataset.target;

      // 更新导航状态
      navItems.forEach(nav => nav.classList.remove('active'));
      this.classList.add('active');

      // 切换内容区域
      sections.forEach(section => {
        section.classList.remove('active');
        if (section.id === target) {
          section.classList.add('active');
        }
      });

      // 平滑滚动到顶部
      window.scrollTo({
        top: document.querySelector('.ai-content-wrapper').offsetTop - 20,
        behavior: 'smooth'
      });
    });
  });

  // 卡片点击动画
  const cards = document.querySelectorAll('.ai-model-card, .ai-tool-card, .ai-app-card');
  cards.forEach(card => {
    card.addEventListener('click', function(e) {
      if (!e.target.closest('a')) {
        this.style.transform = 'scale(0.98)';
        setTimeout(() => {
          this.style.transform = '';
        }, 150);
      }
    });
  });

  // 创建粒子效果
  const particlesContainer = document.getElementById('aiParticles');
  if (particlesContainer) {
    for (let i = 0; i < 20; i++) {
      const particle = document.createElement('div');
      particle.style.cssText = `
        position: absolute;
        width: ${Math.random() * 4 + 2}px;
        height: ${Math.random() * 4 + 2}px;
        background: rgba(255, 255, 255, ${Math.random() * 0.5 + 0.3});
        border-radius: 50%;
        left: ${Math.random() * 100}%;
        top: ${Math.random() * 100}%;
        animation: particleFloat ${Math.random() * 10 + 10}s linear infinite;
        animation-delay: ${Math.random() * -20}s;
      `;
      particlesContainer.appendChild(particle);
    }
  }
})();
</script>
